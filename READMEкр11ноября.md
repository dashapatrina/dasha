```python
import random
import math

def distance_to_centers(point, centers):
    """–í—ã—á–∏—Å–ª—è–µ—Ç —Å—É–º–º—É —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–π –æ—Ç —Ç–æ—á–∫–∏ –¥–æ –≤—Å–µ—Ö —Ü–µ–Ω—Ç—Ä–æ–≤"""
    total_distance = 0
    for center in centers:
        dx = point[0] - center[0]
        dy = point[1] - center[1]
        total_distance += math.sqrt(dx*dx + dy*dy)
    return total_distance

def sa_placement(centers, n_points, temp, cooling):
    points = [[random.uniform(-10, 10), random.uniform(-10, 10)] for _ in range(n_points)]
    current_energy = sum(distance_to_centers(p, centers) for p in points)
    
    while temp > 1e-6:
        # –°–ª—É—á–∞–π–Ω–æ —Å–¥–≤–∏–Ω—É—Ç—å –æ–¥–Ω—É —Ç–æ—á–∫—É
        new_points = [p[:] for p in points]  # –∫–æ–ø–∏—Ä—É–µ–º —Ç–æ—á–∫–∏
        idx = random.randint(0, n_points - 1)
        new_points[idx][0] += random.uniform(-1, 1) * temp
        new_points[idx][1] += random.uniform(-1, 1) * temp
        
        new_energy = sum(distance_to_centers(p, centers) for p in new_points)
        
        # –ü—Ä–∏–Ω—è—Ç—å/–æ—Ç–∫–ª–æ–Ω–∏—Ç—å –Ω–æ–≤—ã–π –Ω–∞–±–æ—Ä –ø–æ –∫—Ä–∏—Ç–µ—Ä–∏—é –ú–µ—Ç—Ä–æ–ø–æ–ª–∏—Å–∞
        if new_energy < current_energy:
            # –ï—Å–ª–∏ –Ω–æ–≤–æ–µ —Ä–µ—à–µ–Ω–∏–µ –ª—É—á—à–µ - –ø—Ä–∏–Ω–∏–º–∞–µ–º
            points = new_points
            current_energy = new_energy
        else:
            # –ï—Å–ª–∏ —Ö—É–∂–µ - –ø—Ä–∏–Ω–∏–º–∞–µ–º —Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é exp(-ŒîE/temp)
            delta_energy = new_energy - current_energy
            probability = math.exp(-delta_energy / temp)
            if random.random() < probability:
                points = new_points
                current_energy = new_energy
        
        temp *= cooling
    
    return points
```
